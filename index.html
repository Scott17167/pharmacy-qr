// ================================================================ // PharmacyQR — Google Apps Script v2.0 // 약국마다 개별 Google Sheets 파일 자동 생성 & 분리 관리 // ================================================================ // // ★ 설정 방법 ★ // 1. 이 코드 전체를 복사 // 2. script.google.com → 기존 프로젝트 열기 // 3. 기존 코드 전체 삭제 후 붙여넣기 // 4. ADMIN_FOLDER_ID 를 내 Google Drive 폴더 ID로 교체 // 5. MASTER_SPREADSHEET_ID 를 기존 마스터 시트 ID로 교체 // 6. 배포 → 새 버전으로 배포 → URL 동일하게 유지 // ================================================================ // ★ 여기만 바꾸세요 ★ // Google Drive에서 약국 시트들을 모아둘 폴더 ID // (Drive에서 폴더 열면 URL에 /folders/XXXX 부분이 ID입니다) const ADMIN_FOLDER_ID = '1CbAOIaV0YAxitY1OVnko28cNGRzv1N7S'; // 전체 현황을 기록할 마스터 시트 ID (기존 그대로 사용) const MASTER_SPREADSHEET_ID = '1wpOmCxkBjVQYP-_1BHPMfX6uUJDZfU5yGFuiA3-qk8Q'; // 컬럼 헤더 const HEADERS = [ '전송시각', '약국코드', '약국명', '성별/연령', '질병', '건강목표', '아침식사', '식성', '흡연', '음주', '운동', '수면', '복용약물요약', '고갈영양소', '추천영양성분', '전체요약' ]; // ================================================================ // POST — 고객이 전송 버튼을 눌렀을 때 // ================================================================ function doPost(e) { try { const data = JSON.parse(e.postData.contents); const pharmacyCode = data.pharmacy_code || 'unknown'; const pharmacyName = data.pharmacy_name || '미등록약국'; const row = [ data.timestamp || new Date().toLocaleString('ko-KR'), pharmacyCode, pharmacyName, (data.gender || '') + '/' + (data.age || ''), data.disease || '없음', data.goal || '없음', data.breakfast || '', data.taste || '', data.smoke || '', data.alcohol || '', data.exercise || '', data.sleep || '', data.summary || '', data.depleted || '', data.recommendations || '', data.summary || '' ]; // 1. 약국 개별 시트에 기록 writeToPharmacySheet(pharmacyCode, pharmacyName, row); // 2. 마스터(전체현황) 시트에도 동시 기록 writeToMasterSheet(row); return ContentService .createTextOutput(JSON.stringify({ status: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ================================================================ // 약국 개별 Sheets 파일에 기록 // - 해당 약국 파일이 없으면 자동 생성 // - Drive 폴더 안에 "[약국코드] 약국명" 이름으로 저장 // ================================================================ function writeToPharmacySheet(pharmacyCode, pharmacyName, row) { const folder = DriveApp.getFolderById(ADMIN_FOLDER_ID); // 이미 해당 약국 파일이 있는지 확인 const fileName = '[' + pharmacyCode + '] ' + pharmacyName; const existingFiles = folder.getFilesByName(fileName); let ss; if (existingFiles.hasNext()) { // 기존 파일 열기 const file = existingFiles.next(); ss = SpreadsheetApp.openById(file.getId()); } else { // 신규 파일 생성 ss = SpreadsheetApp.create(fileName); // 파일을 지정 폴더로 이동 const newFile = DriveApp.getFileById(ss.getId()); folder.addFile(newFile); DriveApp.getRootFolder().removeFile(newFile); // 루트에서 제거 // 헤더 설정 const sheet = ss.getActiveSheet(); sheet.setName('설문결과'); sheet.appendRow(HEADERS); // 헤더 스타일 const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground('#00897b'); headerRange.setFontColor('white'); headerRange.setFontWeight('bold'); headerRange.setFontFamily('Arial'); sheet.setFrozenRows(1); // 열 너비 조정 sheet.setColumnWidth(1, 160); // 전송시각 sheet.setColumnWidth(3, 140); // 약국명 sheet.setColumnWidth(5, 120); // 질병 sheet.setColumnWidth(6, 160); // 건강목표 sheet.setColumnWidth(14, 280); // 고갈영양소 sheet.setColumnWidth(15, 280); // 추천영양성분 // 안내 시트 추가 const infoSheet = ss.insertSheet('📋 약국정보'); infoSheet.appendRow(['약국코드', pharmacyCode]); infoSheet.appendRow(['약국명', pharmacyName]); infoSheet.appendRow(['생성일', new Date().toLocaleString('ko-KR')]); infoSheet.appendRow(['안내', '이 시트는 PharmacyQR 어드민이 관리합니다.']); infoSheet.getRange('A1:B4').setFontFamily('Arial'); infoSheet.getRange('A1:A4').setFontWeight('bold').setFontColor('#00897b'); // 시트 순서 정렬 (설문결과 탭이 앞으로) ss.setActiveSheet(ss.getSheetByName('설문결과')); ss.moveActiveSheet(1); } // 데이터 행 추가 const sheet = ss.getSheetByName('설문결과') || ss.getActiveSheet(); sheet.appendRow(row); // 짝수/홀수 줄 색상 const lastRow = sheet.getLastRow(); if (lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, HEADERS.length).setBackground('#f0f4f3'); } return ss.getId(); } // ================================================================ // 마스터(전체현황) 시트에 기록 // ================================================================ function writeToMasterSheet(row) { try { const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID); let sheet = ss.getSheetByName('📊 전체현황'); if (!sheet) { sheet = ss.insertSheet('📊 전체현황'); sheet.appendRow(HEADERS); const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground('#263238'); headerRange.setFontColor('white'); headerRange.setFontWeight('bold'); sheet.setFrozenRows(1); } sheet.appendRow(row); const lastRow = sheet.getLastRow(); if (lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, HEADERS.length).setBackground('#f0f4f3'); } } catch(err) { // 마스터 시트 오류는 무시 (약국 시트 기록은 성공했으므로) console.log('마스터 시트 기록 오류: ' + err.toString()); } } // ================================================================ // GET — 약국 시트 URL 조회 (어드민용) // 사용법: APPS_SCRIPT_URL?action=getUrl&pharmacy=songpa001 // ================================================================ function doGet(e) { const action = e.parameter.action; const pharmacyCode = e.parameter.pharmacy; if (action === 'getUrl' && pharmacyCode) { const folder = DriveApp.getFolderById(ADMIN_FOLDER_ID); const files = folder.getFilesByName(pharmacyCode); if (files.hasNext()) { const file = files.next(); const ss = SpreadsheetApp.openById(file.getId()); return ContentService .createTextOutput(JSON.stringify({ status: 'ok', url: ss.getUrl(), name: file.getName() })) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({ status: 'not_found' })) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput('PharmacyQR Admin API v2.0 — OK') .setMimeType(ContentService.MimeType.TEXT); }
